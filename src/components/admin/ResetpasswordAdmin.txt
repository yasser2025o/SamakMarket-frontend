<!-- =============================================================
  components/admin/ResetPasswordAdmin.vue
  Reset mot de passe vendeur — recherche + 5 résultats max
============================================================= -->
<template>
  <div class="rpa">

    <!-- Header -->
    <div class="rpa-header">
      <h2 class="rpa-titre">🔑 Reset Mot de Passe Vendeur</h2>
      <p class="rpa-sous">Recherchez un vendeur et réinitialisez son mot de passe</p>
    </div>

    <!-- Barre de recherche -->
    <div class="rpa-search">
      <span class="search-ico">🔍</span>
      <input
        v-model="recherche"
        placeholder="Nom, email ou ville du vendeur..."
        class="search-input"
        @input="filtrerVendeurs"
      />
      <span v-if="recherche" @click="recherche = ''; filtrerVendeurs()" class="search-clear">✕</span>
    </div>

    <!-- Résultats -->
    <div class="rpa-info" v-if="recherche">
      {{ vendeursFiltres.length }} résultat(s) pour "{{ recherche }}"
    </div>

    <!-- Liste vendeurs (max 5) -->
    <div v-if="chargement" class="rpa-loading">
      <span class="spin">⟳</span> Chargement...
    </div>

    <div v-else-if="vendeursFiltres.length === 0 && recherche" class="rpa-vide">
      Aucun vendeur trouvé pour "{{ recherche }}"
    </div>

    <div v-else-if="!recherche" class="rpa-vide">
      Tapez un nom pour rechercher un vendeur
    </div>

    <div v-else class="rpa-liste">
      <div
        v-for="v in vendeursFiltres.slice(0, 5)"
        :key="v.id"
        :class="['rpa-item', selectionne?.id === v.id && 'rpa-item-on']"
        @click="selectionner(v)"
      >
        <div class="rpa-av">{{ v.name?.charAt(0)?.toUpperCase() }}</div>
        <div class="rpa-info-vendeur">
          <div class="rpa-nom">{{ v.name }}</div>
          <div class="rpa-detail">{{ v.email }} · {{ v.city || '—' }}</div>
        </div>
        <div class="rpa-badge">Vendeur</div>
      </div>

      <!-- Indicateur si plus de 5 résultats -->
      <div v-if="vendeursFiltres.length > 5" class="rpa-plus">
        +{{ vendeursFiltres.length - 5 }} autres — affinez la recherche
      </div>
    </div>

    <!-- Formulaire reset -->
    <Transition name="slide">
      <div v-if="selectionne" class="rpa-reset">
        <div class="rpa-reset-titre">
          Réinitialiser le mot de passe de <strong>{{ selectionne.name }}</strong>
        </div>

        <div class="rpa-reset-info">
          Un mot de passe temporaire sera généré et envoyé par email à
          <strong>{{ selectionne.email }}</strong>
        </div>

        <button
          @click="resetPassword"
          :disabled="enCours"
          class="btn-reset"
        >
          <span v-if="enCours" class="spin">⟳</span>
          <span v-else>🔑 Réinitialiser et envoyer par email</span>
        </button>

        <!-- Résultat -->
        <div v-if="resultat" class="rpa-resultat">
          <div class="resultat-ok">✅ {{ resultat.message }}</div>
          <div class="resultat-mdp">
            Mot de passe temporaire :
            <code>{{ resultat.mdpTemp }}</code>
          </div>
        </div>
      </div>
    </Transition>

    <!-- Toast -->
    <Transition name="toast">
      <div v-if="toast.visible" :class="['toast', `toast-${toast.type}`]">
        {{ toast.message }}
      </div>
    </Transition>

  </div>
</template>

<script setup>
import { ref, onMounted } from 'vue'
import api from '@/services/api'

const vendeurs        = ref([])
const vendeursFiltres = ref([])
const recherche       = ref('')
const selectionne     = ref(null)
const chargement      = ref(true)
const enCours         = ref(false)
const resultat        = ref(null)
const toast           = ref({ visible: false, message: '', type: 'ok' })

const afficherToast = (message, type = 'ok') => {
  toast.value = { visible: true, message, type }
  setTimeout(() => { toast.value.visible = false }, 4000)
}

// Filtrer les vendeurs selon la recherche
const filtrerVendeurs = () => {
  selectionne.value = null
  resultat.value    = null

  if (!recherche.value.trim()) {
    vendeursFiltres.value = []
    return
  }

  const q = recherche.value.toLowerCase().trim()
  vendeursFiltres.value = vendeurs.value.filter(v =>
    v.name?.toLowerCase().includes(q) ||
    v.email?.toLowerCase().includes(q) ||
    v.city?.toLowerCase().includes(q)
  )
}

const selectionner = (v) => {
  selectionne.value = v
  resultat.value    = null
}

const resetPassword = async () => {
  if (!selectionne.value) return
  enCours.value = true
  resultat.value = null
  try {
    const { data } = await api.post(`/admin/vendeurs/${selectionne.value.id}/reset-password`)
    resultat.value = data
    afficherToast(`✅ Mot de passe réinitialisé pour ${selectionne.value.name}`, 'ok')
  } catch (err) {
    afficherToast(`❌ ${err.response?.data?.message || 'Erreur'}`, 'err')
  } finally {
    enCours.value = false
  }
}

onMounted(async () => {
  try {
    const { data } = await api.get('/admin/vendeurs')
    vendeurs.value = data.filter(u => u.role === 'seller')
  } catch {
    afficherToast('Erreur chargement vendeurs', 'err')
  } finally {
    chargement.value = false
  }
})
</script>
